210    pub fn into_node_config(self) -> eyre::Result<NodeConfig> {
211        let genesis_balance = Unit::ETHER.wei().saturating_mul(U256::from(self.balance));
212        let compute_units_per_second =
213            if self.evm.no_rate_limit { Some(u64::MAX) } else { self.evm.compute_units_per_second };
214
215        let hardfork = match &self.hardfork {
216            Some(hf) => {
217                if self.evm.optimism {
218                    Some(OptimismHardfork::from_str(hf)?.into())
219                } else {
220                    Some(EthereumHardfork::from_str(hf)?.into())
221                }
222            }
223            None => None,
224        };
225
226        Ok(NodeConfig::default()
227            .with_gas_limit(self.evm.gas_limit)
228            .disable_block_gas_limit(self.evm.disable_block_gas_limit)
229            .with_gas_price(self.evm.gas_price)
230            .with_hardfork(hardfork)
231            .with_blocktime(self.block_time)
232            .with_no_mining(self.no_mining)
233            .with_mixed_mining(self.mixed_mining, self.block_time)
234            .with_account_generator(self.account_generator())
235            .with_genesis_balance(genesis_balance)
236            .with_genesis_timestamp(self.timestamp)
237            .with_port(self.port)
238            .with_fork_choice(match (self.evm.fork_block_number, self.evm.fork_transaction_hash) {
239                (Some(block), None) => Some(ForkChoice::Block(block)),
240                (None, Some(hash)) => Some(ForkChoice::Transaction(hash)),
241                _ => self.evm.fork_url.as_ref().and_then(|f| f.block).map(ForkChoice::Block),
242            })
243            .with_fork_headers(self.evm.fork_headers)
244            .with_fork_chain_id(self.evm.fork_chain_id.map(u64::from).map(U256::from))
245            .fork_request_timeout(self.evm.fork_request_timeout.map(Duration::from_millis))
246            .fork_request_retries(self.evm.fork_request_retries)
247            .fork_retry_backoff(self.evm.fork_retry_backoff.map(Duration::from_millis))
248            .fork_compute_units_per_second(compute_units_per_second)
249            .with_eth_rpc_url(self.evm.fork_url.map(|fork| fork.url))
250            .with_base_fee(self.evm.block_base_fee_per_gas)
251            .disable_min_priority_fee(self.evm.disable_min_priority_fee)
252            .with_storage_caching(self.evm.no_storage_caching)
253            .with_server_config(self.server_config)
254            .with_host(self.host)
255            .set_silent(shell::is_quiet())
256            .set_config_out(self.config_out)
257            .with_chain_id(self.evm.chain_id)
258            .with_transaction_order(self.order)
259            .with_genesis(self.init)
260            .with_steps_tracing(self.evm.steps_tracing)
261            .with_print_logs(!self.evm.disable_console_log)
262            .with_auto_impersonate(self.evm.auto_impersonate)
263            .with_ipc(self.ipc)
264            .with_code_size_limit(self.evm.code_size_limit)
265            .disable_code_size_limit(self.evm.disable_code_size_limit)
266            .set_pruned_history(self.prune_history)
267            .with_init_state(self.load_state.or_else(|| self.state.and_then(|s| s.state)))
268            .with_transaction_block_keeper(self.transaction_block_keeper)
269            .with_max_persisted_states(self.max_persisted_states)
270            .with_optimism(self.evm.optimism)
271            .with_odyssey(self.evm.odyssey)
272            .with_disable_default_create2_deployer(self.evm.disable_default_create2_deployer)
273            .with_slots_in_an_epoch(self.slots_in_an_epoch)
274            .with_memory_limit(self.evm.memory_limit)
275            .with_cache_path(self.cache_path))
276    }

278    fn account_generator(&self) -> AccountGenerator {
279        let mut gen = AccountGenerator::new(self.accounts as usize)
280            .phrase(DEFAULT_MNEMONIC)
281            .chain_id(self.evm.chain_id.unwrap_or_else(|| CHAIN_ID.into()));
282        if let Some(ref mnemonic) = self.mnemonic {
283            gen = gen.phrase(mnemonic);
284        } else if let Some(count) = self.mnemonic_random {
285            let mut rng = rand::thread_rng();
286            let mnemonic = match Mnemonic::<English>::new_with_count(&mut rng, count) {
287                Ok(mnemonic) => mnemonic.to_phrase(),
288                Err(_) => DEFAULT_MNEMONIC.to_string(),
289            };
290            gen = gen.phrase(mnemonic);
291        } else if let Some(seed) = self.mnemonic_seed {
292            let mut seed = StdRng::seed_from_u64(seed);
293            let mnemonic = Mnemonic::<English>::new(&mut seed).to_phrase();
294            gen = gen.phrase(mnemonic);
295        }
296        if let Some(ref derivation) = self.derivation_path {
297            gen = gen.derivation_path(derivation);
298        }
299        gen
300    }

307    /// Starts the node
308    ///
309    /// See also [crate::spawn()]
310    pub async fn run(self) -> eyre::Result<()> {
311        let dump_state = self.dump_state_path();
312        let dump_interval =
313            self.state_interval.map(Duration::from_secs).unwrap_or(DEFAULT_DUMP_INTERVAL);
314        let preserve_historical_states = self.preserve_historical_states;
315
316        let (api, mut handle) = crate::try_spawn(self.into_node_config()?).await?;
317
318        // sets the signal handler to gracefully shutdown.
319        let mut fork = api.get_fork();
320        let running = Arc::new(AtomicUsize::new(0));
321
322        // handle for the currently running rt, this must be obtained before setting the crtlc
323        // handler, See [Handle::current]
324        let mut signal = handle.shutdown_signal_mut().take();
325
326        let task_manager = handle.task_manager();
327        let mut on_shutdown = task_manager.on_shutdown();
328
329        let mut state_dumper =
330            PeriodicStateDumper::new(api, dump_state, dump_interval, preserve_historical_states);
331
332        task_manager.spawn(async move {
333            // wait for the SIGTERM signal on unix systems
334            #[cfg(unix)]
335            let mut sigterm = Box::pin(async {
336                if let Ok(mut stream) =
337                    tokio::signal::unix::signal(tokio::signal::unix::SignalKind::terminate())
338                {
339                    stream.recv().await;
340                } else {
341                    futures::future::pending::<()>().await;
342                }
343            });
344
345            // On windows, this will never fire.
346            #[cfg(not(unix))]
347            let mut sigterm = Box::pin(futures::future::pending::<()>());
348
349            // await shutdown signal but also periodically flush state
350            tokio::select! {
351                 _ = &mut sigterm => {
352                    trace!("received sigterm signal, shutting down");
353                }
354                _ = &mut on_shutdown => {}
355                _ = &mut state_dumper => {}
356            }
357
358            // shutdown received
359            state_dumper.dump().await;
360
361            // cleaning up and shutting down
362            // this will make sure that the fork RPC cache is flushed if caching is configured
363            if let Some(fork) = fork.take() {
364                trace!("flushing cache on shutdown");
365                fork.database
366                    .read()
367                    .await
368                    .maybe_flush_cache()
369                    .expect("Could not flush cache on fork DB");
370                // cleaning up and shutting down
371                // this will make sure that the fork RPC cache is flushed if caching is configured
372            }
373            std::process::exit(0);
374        });
375
376        ctrlc::set_handler(move || {
377            let prev = running.fetch_add(1, Ordering::SeqCst);
378            if prev == 0 {
379                trace!("received shutdown signal, shutting down");
380                let _ = signal.take();
381            }
382        })
383        .expect("Error setting Ctrl-C handler");
384
385        Ok(handle.await??)
386    }

582/// Resolves an alias passed as fork-url to the matching url defined in the rpc_endpoints section
583/// of the project configuration file.
584/// Does nothing if the fork-url is not a configured alias.
585impl AnvilEvmArgs {
586    pub fn resolve_rpc_alias(&mut self) {
587        if let Some(fork_url) = &self.fork_url {
588            if let Ok(config) = Config::load_with_providers(FigmentProviders::Anvil) {
589                if let Some(Ok(url)) = config.get_rpc_url_with_alias(&fork_url.url) {
590                    self.fork_url = Some(ForkUrl { url: url.to_string(), block: fork_url.block });
591                }
592            }
593        }
594    }
595}

597/// Helper type to periodically dump the state of the chain to disk
598struct PeriodicStateDumper {
599    in_progress_dump: Option<Pin<Box<dyn Future<Output = ()> + Send + Sync + 'static>>>,
600    api: EthApi,
601    dump_state: Option<PathBuf>,
602    preserve_historical_states: bool,
603    interval: Interval,
604}
605
606impl PeriodicStateDumper {
607    fn new(
608        api: EthApi,
609        dump_state: Option<PathBuf>,
610        interval: Duration,
611        preserve_historical_states: bool,
612    ) -> Self {
613        let dump_state = dump_state.map(|mut dump_state| {
614            if dump_state.is_dir() {
615                dump_state = dump_state.join("state.json");
616            }
617            dump_state
618        });
619
620        // periodically flush the state
621        let interval = tokio::time::interval_at(Instant::now() + interval, interval);
622        Self { in_progress_dump: None, api, dump_state, preserve_historical_states, interval }
623    }
624
625    async fn dump(&self) {
626        if let Some(state) = self.dump_state.clone() {
627            Self::dump_state(self.api.clone(), state, self.preserve_historical_states).await
628        }
629    }
630
631    /// Infallible state dump
632    async fn dump_state(api: EthApi, dump_state: PathBuf, preserve_historical_states: bool) {
633        trace!(path=?dump_state, "Dumping state on shutdown");
634        match api.serialized_state(preserve_historical_states).await {
635            Ok(state) => {
636                if let Err(err) = foundry_common::fs::write_json_file(&dump_state, &state) {
637                    error!(?err, "Failed to dump state");
638                } else {
639                    trace!(path=?dump_state, "Dumped state on shutdown");
640                }
641            }
642            Err(err) => {
643                error!(?err, "Failed to extract state");
644            }
645        }
646    }
647}
648
649// An endless future that periodically dumps the state to disk if configured.
650impl Future for PeriodicStateDumper {
651    type Output = ();
652
653    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
654        let this = self.get_mut();
655        if this.dump_state.is_none() {
656            return Poll::Pending
657        }
658
659        loop {
660            if let Some(mut flush) = this.in_progress_dump.take() {
661                match flush.poll_unpin(cx) {
662                    Poll::Ready(_) => {
663                        this.interval.reset();
664                    }
665                    Poll::Pending => {
666                        this.in_progress_dump = Some(flush);
667                        return Poll::Pending
668                    }
669                }
670            }
671
672            if this.interval.poll_tick(cx).is_ready() {
673                let api = this.api.clone();
674                let path = this.dump_state.clone().expect("exists; see above");
675                this.in_progress_dump =
676                    Some(Box::pin(Self::dump_state(api, path, this.preserve_historical_states)));
677            } else {
678                break
679            }
680        }
681
682        Poll::Pending
683    }
684}

686/// Represents the --state flag and where to load from, or dump the state to
687#[derive(Clone, Debug)]
688pub struct StateFile {
689    pub path: PathBuf,
690    pub state: Option<SerializableState>,
691}
692
693impl StateFile {
694    /// This is used as the clap `value_parser` implementation to parse from file but only if it
695    /// exists
696    fn parse(path: &str) -> Result<Self, String> {
697        Self::parse_path(path)
698    }
699
700    /// Parse from file but only if it exists
701    pub fn parse_path(path: impl AsRef<Path>) -> Result<Self, String> {
702        let mut path = path.as_ref().to_path_buf();
703        if path.is_dir() {
704            path = path.join("state.json");
705        }
706        let mut state = Self { path, state: None };
707        if !state.path.exists() {
708            return Ok(state)
709        }
710
711        state.state = Some(SerializableState::load(&state.path).map_err(|err| err.to_string())?);
712
713        Ok(state)
714    }
715}

717/// Represents the input URL for a fork with an optional trailing block number:
718/// `http://localhost:8545@1000000`
719#[derive(Clone, Debug, PartialEq, Eq)]
720pub struct ForkUrl {
721    /// The endpoint url
722    pub url: String,
723    /// Optional trailing block
724    pub block: Option<u64>,
725}
726
727impl fmt::Display for ForkUrl {
728    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
729        self.url.fmt(f)?;
730        if let Some(block) = self.block {
731            write!(f, "@{block}")?;
732        }
733        Ok(())
734    }
735}
736
737impl FromStr for ForkUrl {
738    type Err = String;
739
740    fn from_str(s: &str) -> Result<Self, Self::Err> {
741        if let Some((url, block)) = s.rsplit_once('@') {
742            if block == "latest" {
743                return Ok(Self { url: url.to_string(), block: None })
744            }
745            // this will prevent false positives for auths `user:password@example.com`
746            if !block.is_empty() && !block.contains(':') && !block.contains('.') {
747                let block: u64 = block
748                    .parse()
749                    .map_err(|_| format!("Failed to parse block number: `{block}`"))?;
750                return Ok(Self { url: url.to_string(), block: Some(block) })
751            }
752        }
753        Ok(Self { url: s.to_string(), block: None })
754    }
755}

757/// Clap's value parser for genesis. Loads a genesis.json file.
758fn read_genesis_file(path: &str) -> Result<Genesis, String> {
759    foundry_common::fs::read_json_file(path.as_ref()).map_err(|err| err.to_string())
760}

762fn duration_from_secs_f64(s: &str) -> Result<Duration, String> {
763    let s = s.parse::<f64>().map_err(|e| e.to_string())?;
764    if s == 0.0 {
765        return Err("Duration must be greater than 0".to_string());
766    }
767    Duration::try_from_secs_f64(s).map_err(|e| e.to_string())
768}
