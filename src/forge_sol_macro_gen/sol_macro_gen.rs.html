1//! SolMacroGen and MultiSolMacroGen
2//!
3//! This type encapsulates the logic for expansion of a Rust TokenStream from Solidity tokens. It
4//! uses the `expand` method from `alloy_sol_macro_expander` underneath.
5//!
6//! It holds info such as `path` to the ABI file, `name` of the file and the rust binding being
7//! generated, and lastly the `expansion` itself, i.e the Rust binding for the provided ABI.
8//!
9//! It contains methods to read the json abi, generate rust bindings from the abi and ultimately
10//! write the bindings to a crate or modules.
11
12use alloy_sol_macro_expander::expand::expand;
13use alloy_sol_macro_input::{SolInput, SolInputKind};
14use eyre::{Context, OptionExt, Result};
15use foundry_common::fs;
16use proc_macro2::{Span, TokenStream};
17use std::{
18    fmt::Write,
19    path::{Path, PathBuf},
20    str::FromStr,
21};

23pub struct SolMacroGen {
24    pub path: PathBuf,
25    pub name: String,
26    pub expansion: Option<TokenStream>,
27}
28
29impl SolMacroGen {
30    pub fn new(path: PathBuf, name: String) -> Self {
31        Self { path, name, expansion: None }
32    }
33
34    pub fn get_sol_input(&self) -> Result<SolInput> {
35        let path = self.path.to_string_lossy().into_owned();
36        let name = proc_macro2::Ident::new(&self.name, Span::call_site());
37        let tokens = quote::quote! {
38            #name,
39            #path
40        };
41
42        let sol_input: SolInput = syn::parse2(tokens).wrap_err("failed to parse input")?;
43
44        Ok(sol_input)
45    }
46}

48pub struct MultiSolMacroGen {
49    pub artifacts_path: PathBuf,
50    pub instances: Vec<SolMacroGen>,
51}
52
53impl MultiSolMacroGen {
54    pub fn new(artifacts_path: &Path, instances: Vec<SolMacroGen>) -> Self {
55        Self { artifacts_path: artifacts_path.to_path_buf(), instances }
56    }
57
58    pub fn populate_expansion(&mut self, bindings_path: &Path) -> Result<()> {
59        for instance in &mut self.instances {
60            let path = bindings_path.join(format!("{}.rs", instance.name.to_lowercase()));
61            let expansion = fs::read_to_string(path).wrap_err("Failed to read file")?;
62
63            let tokens = TokenStream::from_str(&expansion)
64                .map_err(|e| eyre::eyre!("Failed to parse TokenStream: {e}"))?;
65            instance.expansion = Some(tokens);
66        }
67        Ok(())
68    }
69
70    pub fn generate_bindings(&mut self) -> Result<()> {
71        for instance in &mut self.instances {
72            Self::generate_binding(instance).wrap_err_with(|| {
73                format!(
74                    "failed to generate bindings for {}:{}",
75                    instance.path.display(),
76                    instance.name
77                )
78            })?;
79        }
80
81        Ok(())
82    }
83
84    fn generate_binding(instance: &mut SolMacroGen) -> Result<()> {
85        let input = instance.get_sol_input()?.normalize_json()?;
86
87        let SolInput { attrs: _, path: _, kind } = input;
88
89        let tokens = match kind {
90            SolInputKind::Sol(mut file) => {
91                let sol_attr: syn::Attribute = syn::parse_quote! {
92                    #[sol(rpc, alloy_sol_types = alloy::sol_types, alloy_contract = alloy::contract)]
93                };
94                file.attrs.push(sol_attr);
95                expand(file).wrap_err("failed to expand")?
96            }
97            _ => unreachable!(),
98        };
99
100        instance.expansion = Some(tokens);
101        Ok(())
102    }

104    pub fn write_to_crate(
105        &mut self,
106        name: &str,
107        version: &str,
108        bindings_path: &Path,
109        single_file: bool,
110        alloy_version: Option<String>,
111        alloy_rev: Option<String>,
112    ) -> Result<()> {
113        self.generate_bindings()?;
114
115        let src = bindings_path.join("src");
116
117        let _ = fs::create_dir_all(&src);
118
119        // Write Cargo.toml
120        let cargo_toml_path = bindings_path.join("Cargo.toml");
121        let mut toml_contents = format!(
122            r#"[package]
123name = "{name}"
124version = "{version}"
125edition = "2021"
126
127[dependencies]
128"#
129        );
130
131        let alloy_dep = Self::get_alloy_dep(alloy_version, alloy_rev);
132        write!(toml_contents, "{alloy_dep}")?;
133
134        fs::write(cargo_toml_path, toml_contents).wrap_err("Failed to write Cargo.toml")?;
135
136        let mut lib_contents = String::new();
137        write!(
138            &mut lib_contents,
139            r#"#![allow(unused_imports, clippy::all, rustdoc::all)]
140        //! This module contains the sol! generated bindings for solidity contracts.
141        //! This is autogenerated code.
142        //! Do not manually edit these files.
143        //! These files may be overwritten by the codegen system at any time.
144        "#
145        )?;
146
147        // Write src
148        let parse_error = |name: &str| {
149            format!("failed to parse generated tokens as an AST for {name};\nthis is likely a bug")
150        };
151        for instance in &self.instances {
152            let contents = instance.expansion.as_ref().unwrap();
153
154            let name = instance.name.to_lowercase();
155            let path = src.join(format!("{name}.rs"));
156            let file = syn::parse2(contents.clone())
157                .wrap_err_with(|| parse_error(&format!("{}:{}", path.display(), name)))?;
158            let contents = prettyplease::unparse(&file);
159            if single_file {
160                write!(&mut lib_contents, "{contents}")?;
161            } else {
162                fs::write(path, contents).wrap_err("failed to write to file")?;
163                write_mod_name(&mut lib_contents, &name)?;
164            }
165        }
166
167        let lib_path = src.join("lib.rs");
168        let lib_file = syn::parse_file(&lib_contents).wrap_err_with(|| parse_error("lib.rs"))?;
169        let lib_contents = prettyplease::unparse(&lib_file);
170        fs::write(lib_path, lib_contents).wrap_err("Failed to write lib.rs")?;
171
172        Ok(())
173    }

175    pub fn write_to_module(&mut self, bindings_path: &Path, single_file: bool) -> Result<()> {
176        self.generate_bindings()?;
177
178        let _ = fs::create_dir_all(bindings_path);
179
180        let mut mod_contents = r#"#![allow(unused_imports, clippy::all, rustdoc::all)]
181        //! This module contains the sol! generated bindings for solidity contracts.
182        //! This is autogenerated code.
183        //! Do not manually edit these files.
184        //! These files may be overwritten by the codegen system at any time.
185        "#
186        .to_string();
187
188        for instance in &self.instances {
189            let name = instance.name.to_lowercase();
190            if !single_file {
191                // Module
192                write_mod_name(&mut mod_contents, &name)?;
193                let mut contents = String::new();
194
195                write!(contents, "{}", instance.expansion.as_ref().unwrap())?;
196                let file = syn::parse_file(&contents)?;
197
198                let contents = prettyplease::unparse(&file);
199                fs::write(bindings_path.join(format!("{name}.rs")), contents)
200                    .wrap_err("Failed to write file")?;
201            } else {
202                // Single File
203                let mut contents = String::new();
204                write!(contents, "{}\n\n", instance.expansion.as_ref().unwrap())?;
205                write!(mod_contents, "{contents}")?;
206            }
207        }
208
209        let mod_path = bindings_path.join("mod.rs");
210        let mod_file = syn::parse_file(&mod_contents)?;
211        let mod_contents = prettyplease::unparse(&mod_file);
212
213        fs::write(mod_path, mod_contents).wrap_err("Failed to write mod.rs")?;
214
215        Ok(())
216    }

218    /// Checks that the generated bindings are up to date with the latest version of
219    /// `sol!`.
220    ///
221    /// Returns `Ok(())` if the generated bindings are up to date, otherwise it returns
222    /// `Err(_)`.
223    #[expect(clippy::too_many_arguments)]
224    pub fn check_consistency(
225        &self,
226        name: &str,
227        version: &str,
228        crate_path: &Path,
229        single_file: bool,
230        check_cargo_toml: bool,
231        is_mod: bool,
232        alloy_version: Option<String>,
233        alloy_rev: Option<String>,
234    ) -> Result<()> {
235        if check_cargo_toml {
236            self.check_cargo_toml(name, version, crate_path, alloy_version, alloy_rev)?;
237        }
238
239        let mut super_contents = String::new();
240        write!(
241            &mut super_contents,
242            r#"#![allow(unused_imports, clippy::all, rustdoc::all)]
243            //! This module contains the sol! generated bindings for solidity contracts.
244            //! This is autogenerated code.
245            //! Do not manually edit these files.
246            //! These files may be overwritten by the codegen system at any time.
247            "#
248        )?;
249        if !single_file {
250            for instance in &self.instances {
251                let name = instance.name.to_lowercase();
252                let path = if is_mod {
253                    crate_path.join(format!("{name}.rs"))
254                } else {
255                    crate_path.join(format!("src/{name}.rs"))
256                };
257                let tokens = instance
258                    .expansion
259                    .as_ref()
260                    .ok_or_eyre(format!("TokenStream for {path:?} does not exist"))?
261                    .to_string();
262
263                self.check_file_contents(&path, &tokens)?;
264                write_mod_name(&mut super_contents, &name)?;
265            }
266
267            let super_path =
268                if is_mod { crate_path.join("mod.rs") } else { crate_path.join("src/lib.rs") };
269            self.check_file_contents(&super_path, &super_contents)?;
270        }
271
272        Ok(())
273    }

275    fn check_file_contents(&self, file_path: &Path, expected_contents: &str) -> Result<()> {
276        eyre::ensure!(
277            file_path.is_file() && file_path.exists(),
278            "{} is not a file",
279            file_path.display()
280        );
281        let file_contents = &fs::read_to_string(file_path).wrap_err("Failed to read file")?;
282
283        // Format both
284        let file_contents = syn::parse_file(file_contents)?;
285        let formatted_file = prettyplease::unparse(&file_contents);
286
287        let expected_contents = syn::parse_file(expected_contents)?;
288        let formatted_exp = prettyplease::unparse(&expected_contents);
289
290        eyre::ensure!(
291            formatted_file == formatted_exp,
292            "File contents do not match expected contents for {file_path:?}"
293        );
294        Ok(())
295    }

297    fn check_cargo_toml(
298        &self,
299        name: &str,
300        version: &str,
301        crate_path: &Path,
302        alloy_version: Option<String>,
303        alloy_rev: Option<String>,
304    ) -> Result<()> {
305        eyre::ensure!(crate_path.is_dir(), "Crate path must be a directory");
306
307        let cargo_toml_path = crate_path.join("Cargo.toml");
308
309        eyre::ensure!(cargo_toml_path.is_file(), "Cargo.toml must exist");
310        let cargo_toml_contents =
311            fs::read_to_string(cargo_toml_path).wrap_err("Failed to read Cargo.toml")?;
312
313        let name_check = format!("name = \"{name}\"");
314        let version_check = format!("version = \"{version}\"");
315        let alloy_dep_check = Self::get_alloy_dep(alloy_version, alloy_rev);
316        let toml_consistent = cargo_toml_contents.contains(&name_check) &&
317            cargo_toml_contents.contains(&version_check) &&
318            cargo_toml_contents.contains(&alloy_dep_check);
319        eyre::ensure!(
320            toml_consistent,
321            r#"The contents of Cargo.toml do not match the expected output of the latest `sol!` version.
322                This indicates that the existing bindings are outdated and need to be generated again."#
323        );
324
325        Ok(())
326    }

328    /// Returns the `alloy` dependency string for the Cargo.toml file.
329    /// If `alloy_version` is provided, it will use that version from crates.io.
330    /// If `alloy_rev` is provided, it will use that revision from the GitHub repository.
331    fn get_alloy_dep(alloy_version: Option<String>, alloy_rev: Option<String>) -> String {
332        if let Some(alloy_version) = alloy_version {
333            format!(
334                r#"alloy = {{ version = "{alloy_version}", features = ["sol-types", "contract"] }}"#,
335            )
336        } else if let Some(alloy_rev) = alloy_rev {
337            format!(
338                r#"alloy = {{ git = "https://github.com/alloy-rs/alloy", rev = "{alloy_rev}", features = ["sol-types", "contract"] }}"#,
339            )
340        } else {
341            r#"alloy = { git = "https://github.com/alloy-rs/alloy", features = ["sol-types", "contract"] }"#.to_string()
342        }
343    }
344}

346fn write_mod_name(contents: &mut String, name: &str) -> Result<()> {
347    if syn::parse_str::<syn::Ident>(&format!("pub mod {name};")).is_ok() {
348        write!(contents, "pub mod {name};")?;
349    } else {
350        write!(contents, "pub mod r#{name};")?;
351    }
352    Ok(())
